ההגשה ולאחר האישור בקשת להגשת קודם המתחם את המתעדים רק הוצבו האריזות שלטי, אירועים תבל לטענת שבניגוד מלמדות אלה ראיות המבקש כי נקבע כן(. הסרטונים: להלן; לתגובה2-ו1 נספחים )האישור בקשת שהוגשה לאחר נספח )הסרטונים צולמו שבהם הביקורים במהלך למבקש שהונפקו קבלות להוציא יש (; הראיות שנמחקו הסעיפים כי הטענה הערעור רשות בקשת במוקד. כולם על לא אך, למחוק ולשם, התשובה במסגרת שהועלו ההגנה לטענות ישיר מענה מהווים שהוצאו והראיות בקשר במיוחד אמורים אלה דברים המבקש לטענת. לתשובה התגובה נועדה בדיוק כך כי אירועים תבל טענת לנוכח שכן(, להם שנלווים והתמונות והקבלות )הסרטונים עם משקל עבור לשלם שעליו ללקוח הבהירו שלדבריה האריזות שלטי תלויים היו במתחם שעומדת היא הרלוונטית בתקופה הוצבו אכן האריזות שלטי אם השאלה – האריזות טענות העלתה אירועים שתבל מאחר כי נטען כן. הצדדים שבין המחלוקת בלב היה נדרש, לביסוסן ראיות צירפה ואף האריזות שלטי לעניין פוזיטיביות עובדתיות צירופן את להתיר למצער או בנדון טענתה את שמפריכות ראיות להציג למבקש לאפשר תבל אמינות את לקעקע כדי אף דרושה הסרטונים הצגת כי נטען עוד. חדשות כראיות תום כלפי שהופנו הקשות הטענות את להפריך מנת ועל, מטעמה והמצהירים אירועים להוציא מקום היה לא כי וטוען מוסיף המבקש. המייצגת הכוח ובאת המבקש של הלב שלטי צילומי אמינות את מערערים שלשיטתו הפייסבוק צילומי את התגובה מן את לצלם נהגה אירועים שתבל כך על מעידים הם שכן; לתשובה שצורפו האריזות מספר שצולמו צילומים אותם את רק לצרף בחרה זאת ולמרות, קבוע באופן המתחם שיתר כך על עומד אף המבקש, האמור לצד. האישור בקשת הגשת לאחר חודשים לטענות וישיר ברור מענה מהוות התגובה מן שנמחקו והמשפטיות העובדתיות הטענות . מחיקתן על להורות מקום היה לא כן ועל, ההגנה 3 הועלו ערעור רשות בבקשת שנטענו הטענות כל כי טוענת מצידה אירועים תבל ;
